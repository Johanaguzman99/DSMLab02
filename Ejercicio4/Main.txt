package ejercicio4;

import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class Ejercicio4 {

    public static void main(String[] args) {
        int numPares = 0;
        String puntosCardinales =null;
        String patron = ("(\\([-]?[0-9],[-]?[0-9]\\))");
        Pattern p = Pattern.compile(patron);
        List<String> listaCardinales = new ArrayList<>();
        Scanner reader = new Scanner(System.in);
        
        System.out.print("Inserta el n√∫mero de cordenadas deseadas : ");
        numPares = reader.nextInt();
        
        for(int i=0; i<numPares; i++){
            System.out.print("Ingresa tu coordenada #"+(i+1)+" : ");
            puntosCardinales = reader.next();
            Matcher matcher = p.matcher(puntosCardinales);
            if(matcher.find()){
                listaCardinales.add(puntosCardinales);
            }else{
                System.out.println("Debes seguir la sintaxis correcta!(#,#)");
            }
        }
        
        Coordenadas coord = new Coordenadas(listaCardinales);
        coord.Print();
    }
    
}
