package ejercicio4;

import java.util.ArrayList;
import java.util.List;

public class Coordenadas {
    
    private List<String> coordenadas = new ArrayList<>();
    
    public Coordenadas(List<String> pares){
        coordenadas = pares;
    }
    
    public void Print(){
        String mensaje = "";
        for(int i=0; i <coordenadas.size(); i++){
            String actual = coordenadas.get(i);
            String[] partes = actual.split(",");
            int numX = Integer.valueOf(partes[0].replace("(", ""));
            int numY = Integer.valueOf(partes[1].replace(")", ""));
            
            if(numX ==0 && numY ==0) mensaje+="La coordenada "+actual+" es el origen\n";
            else if(numX >= 0 && numY >= 0) mensaje+= "La coordenada "+actual+" corresponde al primer cuadrante \n";
            else if(numX <= 0 && numY >=0 ) mensaje+= "La coordenada "+actual+" corresponde al segundo cuadrante \n";
            else if(numX <=0 && numY <= 0) mensaje+="La coordenada "+actual+" corresponde al tercer cuadrante\n";
            else if(numX >= 0 && numY <= 0) mensaje+="La coordenada "+actual+"pertenece al cuarto cuadrante\n";
            else {
                System.out.println("Un Error ocurrio!");
                return;
            }   
        }
        System.out.print(mensaje);
    }
}
